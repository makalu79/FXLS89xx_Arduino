22void FXLS89xx_Arduino::run(bool sdcd) {
23 bit_op8(FXLS896x_SENS_CONFIG1, ~FXLS896x_SENS_CONFIG1_ACTIVE_MASK, 0);
24 bit_op8(FXLS896x_SENS_CONFIG1, ~FXLS896x_SENS_CONFIG1_FSR_MASK, sensor_range<<FXLS896x_SENS_CONFIG1_FSR_SHIFT);
25 bit_op8(FXLS896x_SENS_CONFIG2, ~(FXLS896x_SENS_CONFIG2_SLEEP_PM_MASK|FXLS896x_SENS_CONFIG2_WAKE_PM_MASK),
26 (wake_pm<<FXLS896x_SENS_CONFIG2_WAKE_PM_SHIFT) | (sleep_pm<<FXLS896x_SENS_CONFIG2_SLEEP_PM_SHIFT));
27 write_r8(FXLS896x_SENS_CONFIG3, (wake_odr<<FXLS896x_SENS_CONFIG3_WAKE_ODR_SHIFT) | sleep_odr);
28 if (!sdcd)
29 bit_op8(FXLS896x_INT_EN, ~FXLS896x_INT_EN_DRDY_EN_MASK, FXLS896x_INT_EN_DRDY_EN_EN);
30 bit_op8(FXLS896x_SENS_CONFIG1, ~FXLS896x_SENS_CONFIG1_ACTIVE_MASK, FXLS896x_SENS_CONFIG1_ACTIVE_ACTIVE);
31}

33void FXLS89xx_Arduino::sdcd(ref_updm ref, XYZ xyz, float upper_threshold, float lower_threshold, uint8_t debounce) {
34 // Set upper threshold
35 uint16_t upper_threshold_12bit=upper_threshold/0.98/pow(2,sensor_range);
36 if (upper_threshold>1024*pow(2,sensor_range+1)) upper_threshold_12bit=0x07ff;
37 else if (upper_threshold<-1024*pow(2,sensor_range+1)) upper_threshold_12bit=0x0800;
38 Serial.println(upper_threshold_12bit, HEX);
39
40 // Set lower threshold
41 uint16_t lower_threshold_12bit=lower_threshold/0.98/pow(2,sensor_range);
42 if (lower_threshold>1024*pow(2,sensor_range+1)) lower_threshold_12bit=0x07ff;
43 else if (lower_threshold<-1024*pow(2,sensor_range+1)) lower_threshold_12bit=0x0800;
44 Serial.println(lower_threshold_12bit, HEX);
45
46 // Set configuration of debounce counters and lower/upper thresholds
47 uint8_t reg[6] = {debounce, debounce, lower_threshold_12bit&0xff, lower_threshold_12bit>>8, upper_threshold_12bit&0xff, upper_threshold_12bit>>8};
48 reg_w(FXLS896x_SDCD_OT_DBCNT, reg, 6);
49
50 uint8_t sdcd_conf[2] = {
51 xyz<<3|xyz,
52 ref<<FXLS896x_SDCD_CONFIG2_REF_UPDM_SHIFT | FXLS896x_SDCD_CONFIG2_SDCD_EN_EN
53 };
54 reg_w(FXLS896x_SDCD_CONFIG1, sdcd_conf, 2);
55 Serial.println(sdcd_conf[1], HEX);
56
57 // Route SDCD interruption to INT2
58 bit_op8(FXLS896x_INT_EN,
59 ~(FXLS896x_INT_EN_SDCD_WT_EN_MASK | FXLS896x_INT_EN_SDCD_OT_EN_MASK),
60 sdcd_wt ? FXLS896x_INT_EN_SDCD_WT_EN_EN : FXLS896x_INT_EN_SDCD_OT_EN_EN);
61
62 bit_op8(FXLS896x_INT_PIN_SEL,
63 ~(FXLS896x_INT_PIN_SEL_SDCD_WT_INT2_MASK | FXLS896x_INT_PIN_SEL_SDCD_OT_INT2_MASK),
64 sdcd_wt ? FXLS896x_INT_PIN_SEL_SDCD_WT_INT2_EN : FXLS896x_INT_PIN_SEL_SDCD_OT_INT2_EN);
65}

75void FXLS89xx_Arduino::read_XYZ(float *pOutBuffer) {
76 uint8_t output[6];
77 reg_r(FXLS896x_OUT_X_LSB, output, 6);
78 for (int i=0; i<3; i++) {
79 if (output[2*i+1]&0xf8)
80 pOutBuffer[i] = ((float)((((uint8_t)~output[2*i+1])&0x0f)<<8)+((uint8_t)~output[2*i])+1)*(-0.98)*(sensor_range+1);
81 else pOutBuffer[i] = ((float)(((uint8_t)output[2*i+1]&0x0f)<<8)+(uint8_t)output[2*i])*0.98*(sensor_range+1);
82 }
83}
